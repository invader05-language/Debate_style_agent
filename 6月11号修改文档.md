```python
import pytest
from unittest.mock import AsyncMock, patch, MagicMock
from debate.engine import DebateEngine
from debate.protocol import (
    DebateConfig, DebateStatus, Role, Message, Round, Verdict, DebateResult
)

    """Test complete debate flow."""

    @pytest.mark.asyncio
    async def test_full_debate_flow(self, mock_mimo_agent, mock_deepseek_agent):
        """Test complete debate from start to verdict."""
        # Setup mock responses
        mock_mimo_agent.chat.side_effect = [
            "正方论点: JWT 无状态、可扩展",  # Round 1 Pro
            "裁判总结: 双方各有优势",        # Round 1 Judge
            "最终判决: 建议使用 JWT",        # Verdict
        ]
        mock_deepseek_agent.chat.return_value = "反方质疑: JWT 有安全风险"

        # Create engine with mocked agents
        engine = DebateEngine()
        engine.agents = {
            "mimo": mock_mimo_agent,
            "deepseek": mock_deepseek_agent
        }

        # Create config
        config = DebateConfig(
            topic="JWT vs Session",
            max_rounds=1,
            models={"pro": "mimo", "con": "deepseek", "judge": "mimo"}
        )

        # Run debate
        result = await engine.start_debate(config)

        # Verify
        assert result.status == DebateStatus.COMPLETED
        assert len(result.rounds) == 1
        assert result.verdict is not None
        assert result.verdict.recommendation is not None

    @pytest.mark.asyncio
    async def test_debate_early_consensus(self, mock_mimo_agent, mock_deepseek_agent):
        """Test early termination when consensus reached."""
        # Both sides agree
        mock_mimo_agent.chat.side_effect = [
            "我同意反方观点，Session 确实更安全",
            "裁判总结: 双方达成共识",
        ]
        mock_deepseek_agent.chat.return_value = "我同意正方，JWT 也有优势"

        engine = DebateEngine()
        engine.agents = {
            "mimo": mock_mimo_agent,
            "deepseek": mock_deepseek_agent
        }

        config = DebateConfig(
            topic="认证方案",
            max_rounds=3,
            models={"pro": "mimo", "con": "deepseek", "judge": "mimo"}
        )

        result = await engine.start_debate(config)

        # Should terminate early due to consensus
        assert len(result.rounds) < 3

    @pytest.mark.asyncio
    async def test_debate_verdict_json_parsing(self):
        """Test verdict JSON parsing from judge response."""
        engine = DebateEngine()

        # Test valid JSON
        valid_response = '{"recommendation": "Use JWT", "winner": "pro", "confidence": 0.85, "action_plan": ["Step 1"]}'
        verdict = engine._parse_verdict(valid_response, "fallback")
        assert verdict.recommendation == "Use JWT"
        assert verdict.winner == "pro"
        assert verdict.confidence == 0.85

        # Test invalid JSON fallback
        invalid_response = "This is not JSON"
        verdict = engine._parse_verdict(invalid_response, "fallback")
        assert verdict.recommendation == "fallback"
        assert verdict.winner == "draw"

    @pytest.mark.asyncio
    async def test_debate_agent_failure_recovery(self, mock_mimo_agent, mock_deepseek_agent):
        """Test debate continues when one agent fails."""
        # MIMO fails on first call, succeeds on retry

            "mimo": mock_mimo_agent,
```

```sql
-- debate-agent/scripts/init-db.sql
-- Enable pgvector extension
CREATE EXTENSION IF NOT EXISTS vector;

-- Create tables
CREATE TABLE IF NOT EXISTS users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    username VARCHAR(100) UNIQUE NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    hashed_password VARCHAR(255) NOT NULL,
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS debates (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    topic VARCHAR(500) NOT NULL,
    status VARCHAR(20) DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT NOW(),
    completed_at TIMESTAMP,
    verdict JSONB,
    action_plan JSONB
);

CREATE TABLE IF NOT EXISTS messages (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    debate_id UUID REFERENCES debates(id),
    round_number INTEGER,
    role VARCHAR(20),
    content TEXT,
    model_used VARCHAR(50),
    confidence FLOAT,
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS memories (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    topic VARCHAR(500),
    debate_summary TEXT,
    outcome TEXT,
    confidence FLOAT,
    tags TEXT[],
    lessons_learned TEXT[],
    embedding VECTOR(1536),
    created_at TIMESTAMP DEFAULT NOW()
);

CREATE TABLE IF NOT EXISTS executions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    debate_id UUID REFERENCES debates(id),
    status VARCHAR(20),
    code_generated TEXT,
    execution_result TEXT,
    error_message TEXT,
    created_at TIMESTAMP DEFAULT NOW(),
    completed_at TIMESTAMP
);

CREATE TABLE IF NOT EXISTS ai_models (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL,
```

```python
# debate-agent/execution/code_generator.py
"""AI 代码生成器 - 将辩论的 action_plan 转化为可执行代码"""
import json
from typing import List, Optional
from dataclasses import dataclass
from agents.base_agent import BaseAgent


@dataclass
class GeneratedCode:
    """Generated code structure."""
    main_code: str
    test_code: str
    dependencies: List[str]
    language: str


class CodeGenerator:
    """将辩论的 action_plan 转化为可执行代码"""

    def __init__(self, agent: BaseAgent):
        self.agent = agent

    async def generate(self, action_plan: List[str],
                       language: str = "python") -> GeneratedCode:
        """
        生成代码和测试

        Args:
            action_plan: 行动计划步骤
            language: 目标语言 (python/javascript)
        Returns:
            GeneratedCode with main_code, test_code, dependencies
        """
        prompt = f"""根据以下行动计划生成 {language} 代码和单元测试。

行动计划:
{chr(10).join(f'{i+1}. {step}' for i, step in enumerate(action_plan))}

请以 JSON 格式输出:
{{
    "main_code": "主要代码",
    "test_code": "单元测试代码",
    "dependencies": ["依赖包1", "依赖包2"]
}}

确保代码可直接运行，测试使用 pytest。"""

        response = await self.agent.chat(
            system_prompt="你是一个专业的代码生成器，只输出有效的 JSON 格式代码。",
            user_message=prompt
        )

        # Parse JSON response
        try:
            json_start = response.find('{')
            json_end = response.rfind('}') + 1
            if json_start >= 0 and json_end > json_start:
                json_str = response[json_start:json_end]
                data = json.loads(json_str)
                return GeneratedCode(
                    main_code=data.get("main_code", ""),
                    test_code=data.get("test_code", ""),
                    dependencies=data.get("dependencies", []),
                    language=language
                )
        except json.JSONDecodeError:
            pass

        # Fallback: return raw response as code
        return GeneratedCode(
            main_code=response,
            test_code="",
            dependencies=[],
            language=language
        )

    async def refine(self, code: GeneratedCode,
                     error: str) -> GeneratedCode:
        """
        根据执行错误修复代码

        Args:
            code: 原始代码
            error: 错误信息
        Returns:
            Fixed GeneratedCode
        """
        prompt = f"""修复以下代码的错误:

原代码:
{code.main_code}

错误信息:
{error}

请以 JSON 格式输出修复后的代码:
{{
    "main_code": "修复后的代码",
    "test_code": "修复后的测试",
    "dependencies": ["依赖包"]
}}"""

        response = await self.agent.chat(
            system_prompt="你是一个专业的代码修复器，只输出有效的 JSON 格式代码。",
            user_message=prompt
        )

        try:
            json_start = response.find('{')
            json_end = response.rfind('}') + 1
            if json_start >= 0 and json_end > json_start:
                json_str = response[json_start:json_end]
                data = json.loads(json_str)
                return GeneratedCode(
                    main_code=data.get("main_code", code.main_code),
                    test_code=data.get("test_code", code.test_code),
                    dependencies=data.get("dependencies", code.dependencies),
                    language=code.language
                )
        except json.JSONDecodeError:
            pass

        return code
```
